88void delta_motion::run_delta_subprocess() {
89 std::cout << "Running delta program in new console" << std::endl;
90 STARTUPINFO si;
91 PROCESS_INFORMATION pi;
92
93 ZeroMemory(&si, sizeof(si));
94 si.cb = sizeof(si);
95 ZeroMemory(&pi, sizeof(pi));
96 // Start the child process.
97 if (!CreateProcess(pyInterpreter, // No module name (use command line)
98 pyCmd, // Command line
99 NULL, // Process handle not inheritable
100 NULL, // Thread handle not inheritable
101 TRUE, // Set handle inheritance to FALSE
102 CREATE_NEW_CONSOLE, // No creation flags
103 NULL, // Use parent's environment block
104 NULL, // Use parent's starting directory
105 &si, // Pointer to STARTUPINFO structure
106 &pi) // Pointer to PROCESS_INFORMATION structure
107 )
108 {
109 printf("CreateProcess failed (%d).\n", GetLastError());
110 return;
111 }
112
113 // Wait until child process exits.
114 WaitForSingleObject(pi.hProcess, NULL);
115
116 // Close process and thread handles.
117 CloseHandle(pi.hProcess);
118 CloseHandle(pi.hThread);
