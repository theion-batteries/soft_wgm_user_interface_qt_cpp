192void whs_controller::monitor_and_calibrate()
193{
194 std::cout << "monitor calibration started" << std::endl;
195 // algorithm: if current distance is not equal desired thickness, calibrate
196 // 1. check if diff is a value that is below the step resolution, continue or
197 // 2. move up if diff negativ
198 // 3. move down if diff positiv
199 while (true)
200 {
201 double diff = distSensor->getMesuredValue() - _whs_params.thickness;
202 if (abs(diff) <= _whs_params.mm_step_res) // if the difference mesured lower than min step res, we skip calibration
203 {
204 continue;
205 }
206 else
207 {
208 if (diff < 0) // if diff negativ, we move up
209 {
210 linearMover->move_up_by(diff);
211 }
212 else
213 {
214 linearMover->move_down_by(diff);
215
216 }
217
218 }
