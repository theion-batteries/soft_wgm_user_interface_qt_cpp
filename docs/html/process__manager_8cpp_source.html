154void wgm_processes::process_manager::start_all(std::function<void(int)> callback)
155{
156 proc_time = 0;
157 std::cout << "executing all processes" << std::endl;
158 for (auto process : processesvector)
159 {
160 if (process != nullptr)
161 {
162 if (process->start_process() == wgm_feedbacks::enum_proc_feedback::proc_error)
163 {
164 callback(process->get_id());
165 proc_time += process->get_elapsed_time();
166 break;
167 }
168 callback(process->get_id());
169 proc_time += process->get_elapsed_time();
170 }
171 else std::cout << "empty process scheduler" << std::endl;
172 }
173}
